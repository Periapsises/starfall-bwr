local fullCoreDisplayTextureInitialized = false

local indicatorTextureWidth = 64
local indicatorTextureHeight = 32

local function initializeFullCoreDisplayTexture()
    if fullCoreDisplayTextureInitialized then return end
    fullCoreDisplayTextureInitialized = true

    render.selectRenderTarget( "FullCoreDisplay" )

    local xOffset = 0
    local yOffset = 0

    local function drawIndicator( color, text, text2 )
        render.setColor( color )
        render.drawRect( xOffset, yOffset, indicatorTextureWidth, indicatorTextureHeight )

        render.setColor( black )

        if text2 then
            render.drawSimpleText( xOffset + indicatorTextureWidth / 2, yOffset + indicatorTextureHeight * 0.3, text, TEXT_ALIGN.CENTER, TEXT_ALIGN.CENTER )
            render.drawSimpleText( xOffset + indicatorTextureWidth / 2, yOffset + indicatorTextureHeight * 0.7, text2, TEXT_ALIGN.CENTER, TEXT_ALIGN.CENTER )
        else
            render.drawSimpleText( xOffset + indicatorTextureWidth / 2, yOffset + indicatorTextureHeight / 2, text, TEXT_ALIGN.CENTER, TEXT_ALIGN.CENTER )
        end

        xOffset = xOffset + indicatorTextureWidth
        if xOffset > 1024 - indicatorTextureWidth then
            xOffset = 0
            yOffset = yOffset + indicatorTextureHeight
        end
    end

    for _, rodData in pairs( FullCoreDisplay.RodIndicators ) do
        drawIndicator( inactiveColor, rodData.Name )
        drawIndicator( white, rodData.Name )
    end

    drawIndicator( inactiveColor, "FULL", "IN" )
    drawIndicator( fullInColor, "FULL", "IN" )
    drawIndicator( inactiveColor, "FULL", "OUT" )
    drawIndicator( fullOutColor, "FULL", "OUT" )

    drawIndicator( inactiveColor, "ACCUM" )
    drawIndicator( accumColor, "ACCUM" )
    drawIndicator( inactiveColor, "DRIFT" )
    drawIndicator( driftColor, "DRIFT" )
    drawIndicator( inactiveColor, "SCRAM" )
    drawIndicator( scramColor, "SCRAM" )

    drawIndicator( lprmOffColor, "LPRM A", "DOWN" )
    drawIndicator( lprmOffColor, "LPRM B", "DOWN" )
    drawIndicator( lprmOffColor, "LPRM C", "DOWN" )
    drawIndicator( lprmOffColor, "LPRM D", "DOWN" )

    drawIndicator( lprmOnColor, "LPRM A", "DOWN" )
    drawIndicator( lprmOnColor, "LPRM B", "DOWN" )
    drawIndicator( lprmOnColor, "LPRM C", "DOWN" )
    drawIndicator( lprmOnColor, "LPRM D", "DOWN" )

    drawIndicator( lprmOffColor, "LPRM A", "UP" )
    drawIndicator( lprmOffColor, "LPRM B", "UP" )
    drawIndicator( lprmOffColor, "LPRM C", "UP" )
    drawIndicator( lprmOffColor, "LPRM D", "UP" )

    drawIndicator( lprmOnColor, "LPRM A", "UP" )
    drawIndicator( lprmOnColor, "LPRM B", "UP" )
    drawIndicator( lprmOnColor, "LPRM C", "UP" )
    drawIndicator( lprmOnColor, "LPRM D", "UP" )

    render.selectRenderTarget()
end

local fullCoreDisplayMesh
local fullCoreDisplayMeshInitialized = false

local indicatorWidth = 16
local indicatorHeight = 8
local indicatorSpacing = 1

local indicatorGroupSize = ( indicatorWidth + indicatorSpacing ) * 2

local sIndicatorIndex = #FullCoreDisplay.RodIndicators * 2 + 1

local function initializeFullCoreDisplayMesh()
    if fullCoreDisplayMeshInitialized then return end
    fullCoreDisplayMeshInitialized = true

    local vertices = {}

    local function insertVerts( v1, v2, v3, v4 )
        table.insert( vertices, v1 )
        table.insert( vertices, v2 )
        table.insert( vertices, v3 )

        table.insert( vertices, v3 )
        table.insert( vertices, v4 )
        table.insert( vertices, v1 )
    end

    local indicatorsPerRow = 1024 / indicatorTextureWidth
    local indicatorsPerColumn = 1024 / indicatorTextureHeight

    local function setVertexUV( v1, v2, v3, v4, indicatorIndex )
        local U1 = ( indicatorIndex - 1 ) % indicatorsPerRow
        local V1 = math.floor( ( indicatorIndex - 1 ) / indicatorsPerRow )
        local U2 = U1 + 1
        local V2 = V1 + 1

        v1.u = U1 / indicatorsPerRow
        v1.v = V1 / indicatorsPerColumn

        v2.u = U2 / indicatorsPerRow
        v2.v = V1 / indicatorsPerColumn

        v3.u = U2 / indicatorsPerRow
        v3.v = V2 / indicatorsPerColumn

        v4.u = U1 / indicatorsPerRow
        v4.v = V2 / indicatorsPerColumn
    end

    local function getRodIndicatorVerts( x, y, indicatorPositioning )
        local xOffset = indicatorPositioning.x * indicatorWidth + indicatorPositioning.xOffset * indicatorSpacing
        local yOffset = indicatorPositioning.y * indicatorHeight + indicatorPositioning.yOffset * indicatorSpacing

        local v1 = { pos = Vector( x + xOffset, y + yOffset, 0 ) }
        local v2 = { pos = Vector( x + xOffset + indicatorWidth, y + yOffset, 0 ) }
        local v3 = { pos = Vector( x + xOffset + indicatorWidth, y + yOffset + indicatorHeight, 0 ) }
        local v4 = { pos = Vector( x + xOffset, y + yOffset + indicatorHeight, 0 ) }

        return v1, v2, v3, v4
    end

    local function drawRodDisplayData( index, displayData )
        local x = ( -7 + displayData.Position.x ) * indicatorGroupSize
        local y = ( 7 - displayData.Position.y ) * indicatorGroupSize

        local v1, v2, v3, v4

        v1, v2, v3, v4 = getRodIndicatorVerts( x, y, displayData.Positioning.FullInIndicator )
        insertVerts( v1, v2, v3, v4 )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 1 )

        v1, v2, v3, v4 = getRodIndicatorVerts( x, y, displayData.Positioning.FullOutIndicator )
        insertVerts( v1, v2, v3, v4 )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 2 )

        v1, v2, v3, v4 = getRodIndicatorVerts( x, y, displayData.Positioning.SelectedIndicator )
        setVertexUV( v1, v2, v3, v4, ( index * 2 ) - 1 )
        insertVerts( v1, v2, v3, v4 )

        v1, v2, v3, v4 = getRodIndicatorVerts( x, y, displayData.Positioning.DriftIndicator )
        insertVerts( v1, v2, v3, v4 )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 6 )

        v1, v2, v3, v4 = getRodIndicatorVerts( x, y, displayData.Positioning.AccumIndicator )
        insertVerts( v1, v2, v3, v4 )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 4 )

        v1, v2, v3, v4 = getRodIndicatorVerts( x, y, displayData.Positioning.ScramIndicator )
        insertVerts( v1, v2, v3, v4 )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 8 )

    end

    -- LPRM Indicators start on the bottom left corner after three rod indicators and are spaced with two rod indicators.

    local lprmIndicatorOffset = ( indicatorWidth + indicatorSpacing ) * 3
    local lprmIndicatorGroupSize = ( indicatorWidth + indicatorSpacing ) * 4

    local function getLPRMIndicatorVerts( x, y )
        local v1 = { pos = Vector( x, y, 0 ) }
        local v2 = { pos = Vector( x + indicatorWidth, y, 0 ) }
        local v3 = { pos = Vector( x + indicatorWidth, y + indicatorHeight, 0 ) }
        local v4 = { pos = Vector( x, y + indicatorHeight, 0 ) }

        return v1, v2, v3, v4
    end

    local function drawLPRMDisplayData( displayData )
        local x = ( -3.5 + displayData.Position.x ) * lprmIndicatorGroupSize + lprmIndicatorOffset
        local y = ( 3.5 - displayData.Position.y ) * lprmIndicatorGroupSize - lprmIndicatorOffset

        local v1, v2, v3, v4

        v1, v2, v3, v4 = getLPRMIndicatorVerts( x, y )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 17 )
        insertVerts( v1, v2, v3, v4 )

        v1, v2, v3, v4 = getLPRMIndicatorVerts( x, y + indicatorHeight )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 16 )
        insertVerts( v1, v2, v3, v4 )

        v1, v2, v3, v4 = getLPRMIndicatorVerts( x, y + indicatorHeight * 2 + indicatorSpacing )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 15 )
        insertVerts( v1, v2, v3, v4 )

        v1, v2, v3, v4 = getLPRMIndicatorVerts( x, y + indicatorHeight * 3 + indicatorSpacing )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 14 )
        insertVerts( v1, v2, v3, v4 )

        v1, v2, v3, v4 = getLPRMIndicatorVerts( x + indicatorWidth + indicatorSpacing, y )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 21 )
        insertVerts( v1, v2, v3, v4 )

        v1, v2, v3, v4 = getLPRMIndicatorVerts( x + indicatorWidth + indicatorSpacing, y + indicatorHeight )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 20 )
        insertVerts( v1, v2, v3, v4 )

        v1, v2, v3, v4 = getLPRMIndicatorVerts( x + indicatorWidth + indicatorSpacing, y + indicatorHeight * 2 + indicatorSpacing )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 19 )
        insertVerts( v1, v2, v3, v4 )

        v1, v2, v3, v4 = getLPRMIndicatorVerts( x + indicatorWidth + indicatorSpacing, y + indicatorHeight * 3 + indicatorSpacing )
        setVertexUV( v1, v2, v3, v4, sIndicatorIndex + 18 )
        insertVerts( v1, v2, v3, v4 )
    end

    for index, rodDisplayData in ipairs( FullCoreDisplay.RodIndicators ) do
        drawRodDisplayData( index, rodDisplayData )
    end

    for _, lprmDisplayData in ipairs( FullCoreDisplay.LPRMIndicators ) do
        drawLPRMDisplayData( lprmDisplayData )
    end

    fullCoreDisplayMesh = mesh.createFromTable( vertices )
end

FullCoreDisplay.Model = hologram.create( Vector(), Angle(), "models/hunter/blocks/cube025x025x025.mdl", Vector( 1, 1, 1 ) )
FullCoreDisplay.Model:setNoDraw( true )

local function initializeFullCoreDisplay()
    initializeFullCoreDisplayTexture()
    initializeFullCoreDisplayMesh()

    FullCoreDisplay.Model:setMaterial( "!" .. fullCoreDisplayMaterial:getName() )
    FullCoreDisplay.Model:setMesh( fullCoreDisplayMesh )

    hook.remove( "RenderOffScreen", "InitializeFullCoreDisplay" )
end

hook.add( "RenderOffScreen", "InitializeFullCoreDisplay", initializeFullCoreDisplay )
